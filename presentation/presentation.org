  (+ a b))
(add 2 2)
#+end_src

#+latex: \pause
#+begin_src python
  __start: PUSH 2      # Połóż "a" na stosie.
           PUSH 2
           ENTER add   # Wejdź do domknięcia "add".
  add:     TAKE 2      # Pobierz dwa argumenty ze stosu.
           PUSH __add0
           ENTER 1     # PUSHC 2, RETURN
  __add0:  PUSH __add1
           ENTER 0
  __add1:  OP_ADD
           RETURN
#+end_src
