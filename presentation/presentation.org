** Przykład kodu bajtowego
#+latex: \begin{multicols}{2}

#+begin_src scheme
(define (add a b)
  (+ a b))
(add 2 2)
#+end_src

#+latex: \columnbreak
#+latex: \pause
#+begin_src python
  __start: NEXT 2
           NEXT 2
           ENTER add
  add:     TAKE
           TAKE
           NEXT __add0
           ENTER 1
  __add0:  NEXT __add1
           ENTER 0
  __add1:  PRIMOP '+'
           RETURN
#+end_src

#+latex: \end{multicols}
